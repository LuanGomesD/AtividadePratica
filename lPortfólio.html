# Função para escolha de serviço
def escolha_servico():
    while True:  # Laço infinito para garantir uma entrada válida
        print('\nEntre com o tipo de serviço desejado')
        print('DIG - Digitalização')
        print('ICO - Impressão Colorida')
        print('IPB - Impressão Preto e Branco')
        print('FOT - Fotocópia')
        servico = input('>>').upper().strip()  # Converte para maiúsculas e remove espaços
        if servico == 'DIG':
            return 1.10  # Retorna o valor correspondente
        elif servico == 'ICO':
            return 1.00  # Retorna o valor correspondente
        elif servico == 'IPB':
            return 0.40  # Retorna o valor correspondente
        elif servico == 'FOT':
            return 0.20  # Retorna o valor correspondente
        else:
            print('Escolha inválida, entre com o tipo do serviço novamente')  # Mensagem de erro
            continue  # Volta ao início do laço

# Função para número de páginas
def num_pagina():
    while True:
        try:
            pagina = int(input('\nEntre com o número de páginas: '))
            if pagina >= 20000:
                print('Não aceitamos tantas páginas de uma vez.\nPor favor, entre com o número de páginas novamente.')  # Mensagem de erro
                continue  # Retorna ao início do laço
            elif 20 <= pagina < 200:
                descontopagina = pagina * 0.85  # 15% de desconto
                return descontopagina
            elif 200 <= pagina < 2000:
                descontopagina = pagina * 0.80  # 20% de desconto
                return descontopagina
            elif 2000 <= pagina < 20000:
                descontopagina = pagina * 0.75  # 25% de desconto
                return descontopagina
            else:
                return pagina  # Retorna o número de páginas sem desconto
        except ValueError:
            print('Insira um número inteiro')  # Mensagem de erro para entrada inválida

# Função para serviços extras
def servico_extra():
    while True:
        print('\nDeseja adicionar algum serviço?')
        print('1 - Encadernação Simples - R$ 15.00')
        print('2 - Encadernação Capa Dura - R$ 40.00')
        print('0 - Não desejo mais nada')
        extra = input('>>')
        if extra == '0':
            return 0  # Retorna o valor correspondente
        elif extra == '1':
            return 15  # Retorna o valor correspondente
        elif extra == '2':
            return 40  # Retorna o valor correspondente
        else:
            print('Insira uma opção válida')  # Mensagem de erro
            continue  # Volta ao início do laço

# Código principal (main)
def main():
    boas_vindas()
    serv = escolha_servico()
    pg = num_pagina()
    ex = servico_extra()
    total = (serv * pg) + ex  # Calcula o total a pagar

    print(f'\nTotal: R$ {total:.2f} (serviço: {serv:.2f} * páginas: {pg:.0f} + extra: {ex:.2f})')

# Executa o código principal
if __name__ == '__main__':
    main()

            # Variáveis globais
lista_livro = []
id_global = 0

# Função para cadastrar livro
def cadastrar_livro(id):
    print('----- MENU CADASTRAR LIVRO -----')
    print('Id do livro: {}'.format(id))
    nome = input('Por favor entre com o nome do livro: ')
    autor = input('Por favor entre com o autor do livro: ')
    editora = input('Por favor entre com a editora do livro: ')
    dicionario_livro = {
        'Id': id,
        'Nome': nome,
        'Autor': autor,
        'Editora': editora
    }
    lista_livro.append(dicionario_livro.copy())

# Função para consultar livro
def consultar_livro():
    while True:
        print('----- MENU CONSULTAR LIVRO -----')
        menu_consultar = input('Escolha a opção desejada:\n' +
                               '1 - Consultar Todos\n' +
                               '2 - Consultar por Id\n' +
                               '3 - Consultar por Autor\n' +
                               '4 - Retornar ao menu\n >> ')
        if menu_consultar == '1':
            print('Você escolheu consultar todos os livros')
            for livro in lista_livro:
                for key, value in livro.items():
                    print('{}: {}'.format(key, value))
            print('---------------------------------------')
        elif menu_consultar == '2':
            id_livro = int(input('Informe o Id do Livro desejado >>> '))
            for livro in lista_livro:
                if livro['Id'] == id_livro:
                    for key, value in livro.items():
                        print('{}: {}'.format(key, value))
            print('---------------------------------------')
        elif menu_consultar == '3':
            autor_livro = input('Informe o autor do Livro desejado >>> ')
            for livro in lista_livro:
                if livro['Autor'] == autor_livro:
                    for key, value in livro.items():
                        print('{}: {}'.format(key, value))
            print('---------------------------------------')
        elif menu_consultar == '4':
            print('Você escolheu retornar ao menu principal')
            return
        else:
            print('Opção inválida, tente novamente.')

# Função para remover livro
def remover_livro():
    print('----- MENU REMOVER LIVRO -----')
    menu_remover = int(input('Informe o Id do livro que deseja remover >>> '))
    for livro in lista_livro:
        if livro['Id'] == menu_remover:
            lista_livro.remove(livro)
            print('Livro removido com sucesso.')
            return
    print('Id inválido, tente novamente.')

# Função principal
print('Bem vindo a Livraria do Luan Gomes Duarte RU: 4698800 ')
while True:
    print('----- MENU PRINCIPAL -----')
    menu_principal = input('Escolha a opção desejada:\n' +
                           '1 - Cadastrar Livro\n' +
                           '2 - Consultar Livro(s)\n' +
                           '3 - Remover Livro\n' +
                           '4 - Sair\n >> ')
    if menu_principal == '1':
        id_global += 1
        cadastrar_livro(id_global)
    elif menu_principal == '2':
        consultar_livro()
    elif menu_principal == '3':
        remover_livro()
    elif menu_principal == '4':
        break
    else:
        print('Opção inválida, tente novamente.')
